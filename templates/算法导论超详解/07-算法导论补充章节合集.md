## 第 24 章：字符串匹配（String Matching）⭐⭐⭐⭐⭐

### 24.1 问题定义

```
输入：
- 文本串 T[1..n]
- 模式串 P[1..m]，m ≤ n

输出：
所有有效位移 s，使得 T[s+1..s+m] = P[1..m]

应用：
- 文本编辑器查找
- DNA 序列分析
- 搜索引擎
- 入侵检测
```

---

### 24.2 朴素算法 O(mn)

```javascript
/**
 * 朴素字符串匹配
 * 时间复杂度：O(mn)
 */
function naiveMatcher(T, P) {
    const n = T.length;
    const m = P.length;
    const matches = [];
    
    for (let s = 0; s <= n - m; s++) {
        let match = true;
        
        for (let j = 0; j < m; j++) {
            if (T[s + j] !== P[j]) {
                match = false;
                break;
            }
        }
        
        if (match) {
            matches.push(s);
        }
    }
    
    return matches;
}

// 测试
const T = "ababcabcabababd";
const P = "ababd";
console.log(naiveMatcher(T, P));  // [10]
```

### 24.3 Rabin-Karp 算法⭐⭐⭐⭐

#### 算法思想
```
使用哈希函数：
1. 计算模式串的哈希值 h(P)
2. 计算文本串每个长度为 m 的子串的哈希值
3. 只有哈希值相等时才进行字符比较

关键技巧：
- 滚动哈希：O(1) 计算下一个子串的哈希

时间复杂度：
- 平均：O(n+m)
- 最坏：O(nm)（很少发生）
```

#### 实现
```javascript
/**
 * Rabin-Karp 字符串匹配
 * 使用滚动哈希
 */
function rabinKarp(T, P, q = 101) {
    const n = T.length;
    const m = P.length;
    const matches = [];
    
    if (m > n) return matches;
    
    const d = 256;  // 字母表大小
    const h = Math.pow(d, m - 1) % q;
    
    // 计算初始哈希
    let pHash = 0;  // 模式串哈希
    let tHash = 0;  // 文本串当前窗口哈希
    
    for (let i = 0; i < m; i++) {
        pHash = (d * pHash + P.charCodeAt(i)) % q;
        tHash = (d * tHash + T.charCodeAt(i)) % q;
    }
    
    // 滑动窗口
    for (let s = 0; s <= n - m; s++) {
        // 哈希值相等，检查是否真的匹配
        if (pHash === tHash) {
            let match = true;
            
            for (let j = 0; j < m; j++) {
                if (T[s + j] !== P[j]) {
                    match = false;
                    break;
                }
            }
            
            if (match) {
                matches.push(s);
            }
        }
        
        // 滚动哈希（计算下一个窗口的哈希）
        if (s < n - m) {
            tHash = (d * (tHash - T.charCodeAt(s) * h) + 
                     T.charCodeAt(s + m)) % q;
            
            // 处理负数
            if (tHash < 0) {
                tHash += q;
            }
        }
    }
    
    return matches;
}

// 测试
console.log(rabinKarp(T, P));  // [10]
```

### 24.4 KMP 算法⭐⭐⭐⭐⭐高频考点

#### 算法思想
```
Knuth-Morris-Pratt 算法

核心：预处理模式串，构建部分匹配表（π函数）
      避免不必要的回溯

π[i] = P[0..i] 的最长公共前后缀的长度

时间复杂度：
- 预处理：O(m)
- 匹配：O(n)
- 总计：O(n+m)
```

#### 构建部分匹配表
```javascript
/**
 * 构建 KMP 的部分匹配表（前缀函数）
 */
function computePrefixFunction(P) {
    const m = P.length;
    const pi = new Array(m).fill(0);
    
    let k = 0;  // 当前最长公共前后缀长度
    
    for (let q = 1; q < m; q++) {
        // 不匹配时回退
        while (k > 0 && P[k] !== P[q]) {
            k = pi[k - 1];
        }
        
        // 匹配则延长
        if (P[k] === P[q]) {
            k++;
        }
        
        pi[q] = k;
    }
    
    return pi;
}

// 示例：P = "ababaca"
// π = [0, 0, 1, 2, 3, 0, 1]
```

#### KMP 匹配器
```javascript
/**
 * KMP 字符串匹配
 */
function kmpMatcher(T, P) {
    const n = T.length;
    const m = P.length;
    const matches = [];
    
    const pi = computePrefixFunction(P);
    
    let q = 0;  // 已匹配的字符数
    
    for (let i = 0; i < n; i++) {
        // 不匹配时回退
        while (q > 0 && P[q] !== T[i]) {
            q = pi[q - 1];
        }
        
        // 匹配则前进
        if (P[q] === T[i]) {
            q++;
        }
        
        // 完全匹配
        if (q === m) {
            matches.push(i - m + 1);
            q = pi[q - 1];  // 继续找下一个匹配
        }
    }
    
    return matches;
}

// 详细执行过程
function kmpWithTrace(T, P) {
    const pi = computePrefixFunction(P);
    console.log('模式串:', P);
    console.log('π数组:', pi);
    
    let q = 0;
    for (let i = 0; i < T.length; i++) {
        while (q > 0 && P[q] !== T[i]) {
            console.log(`位置 ${i}: '${P[q]}' ≠ '${T[i]}', 回退到 ${pi[q-1]}`);
            q = pi[q - 1];
        }
        
        if (P[q] === T[i]) {
            q++;
            console.log(`位置 ${i}: 匹配，q=${q}`);
        }
        
        if (q === P.length) {
            console.log(`找到匹配！起始位置：${i - P.length + 1}`);
            q = pi[q - 1];
        }
    }
}

// 测试
kmpWithTrace("ababcabcabababd", "ababd");
```

**KMP 的优势：**
```
vs 朴素算法：
✓ 不回溯文本串指针
✓ 利用已匹配信息
✓ 保证 O(n+m) 时间

vs Rabin-Karp：
✓ 没有哈希冲突问题
✓ 更适合小字母表
```

## 第 25 章：多项式与 FFT⭐⭐⭐⭐

### 25.1 多项式表示

```
系数表示：A(x) = a₀ + a₁x + ... + aₙ₋₁xⁿ⁻¹
点值表示：{(x₀, y₀), (x₁, y₁), ..., (xₙ₋₁, yₙ₋₁)}

乘法复杂度：
- 系数表示：O(n²)
- 点值表示：O(n)

FFT 的作用：
在 O(n log n) 时间内完成系数↔点值的转换
```

---

### 25.2 FFT 算法

```javascript
/**
 * 快速傅里叶变换（FFT）
 * 递归实现
 */
function fft(a, invert) {
    const n = a.length;
    
    if (n === 1) return a;
    
    // 分治：奇偶分开
    const a0 = [], a1 = [];
    for (let i = 0; i < n / 2; i++) {
        a0.push(a[2 * i]);
        a1.push(a[2 * i + 1]);
    }
    
    // 递归
    fft(a0, invert);
    fft(a1, invert);
    
    // 合并
    const angle = (2 * Math.PI / n) * (invert ? -1 : 1);
    const wn = { re: Math.cos(angle), im: Math.sin(angle) };
    let w = { re: 1, im: 0 };
    
    for (let i = 0; i < n / 2; i++) {
        const t = {
            re: w.re * a1[i].re - w.im * a1[i].im,
            im: w.re * a1[i].im + w.im * a1[i].re
        };
        
        a[i] = {
            re: a0[i].re + t.re,
            im: a0[i].im + t.im
        };
        
        a[i + n / 2] = {
            re: a0[i].re - t.re,
            im: a0[i].im - t.im
        };
        
        // w *= wn
        const newW = {
            re: w.re * wn.re - w.im * wn.im,
            im: w.re * wn.im + w.im * wn.re
        };
        w = newW;
    }
}

/**
 * 多项式乘法（使用 FFT）
 * 时间复杂度：O(n log n)
 */
function multiplyPolynomials(A, B) {
    const n = A.length + B.length - 1;
    const size = 1;
    while (size < n) size *= 2;
    
    // 补齐到 2 的幂
    const fa = A.map(x => ({ re: x, im: 0 }));
    const fb = B.map(x => ({ re: x, im: 0 }));
    
    while (fa.length < size) fa.push({ re: 0, im: 0 });
    while (fb.length < size) fb.push({ re: 0, im: 0 });
    
    // FFT
    fft(fa, false);
    fft(fb, false);
    
    // 点值相乘
    for (let i = 0; i < size; i++) {
        const temp = {
            re: fa[i].re * fb[i].re - fa[i].im * fb[i].im,
            im: fa[i].re * fb[i].im + fa[i].im * fb[i].re
        };
        fa[i] = temp;
    }
    
    // IFFT
    fft(fa, true);
    
    // 归一化并取实部
    const result = fa.slice(0, n).map(x => 
        Math.round(x.re / size)
    );
    
    return result;
}

// 测试
const poly1 = [1, 2, 1];  // 1 + 2x + x²
const poly2 = [1, 1];     // 1 + x
console.log(multiplyPolynomials(poly1, poly2));
// 输出：[1, 3, 3, 1]  // 1 + 3x + 3x² + x³
```

## 第 26 章：NP 完全性（NP-Completeness）⭐⭐⭐⭐⭐重点

### 26.1 P、NP、NPC 问题

#### 定义
```
P 类问题（Polynomial time）：
- 能在多项式时间内求解的问题
- 例如：排序、最短路径、MST

NP 类问题（Nondeterministic Polynomial time）：
- 能在多项式时间内验证解的问题
- 包含 P 类问题
- 例如：旅行商、背包问题、SAT

NP-Complete 问题：
- NP 中最难的一类问题
- 任何一个 NPC 问题有多项式解 ⇒ 所有 NP 问题都有
- 例如：SAT、3-CNF-SAT、团问题、顶点覆盖

NP-Hard 问题：
- 至少和 NPC 一样难
- 不一定是 NP 问题
```

#### 关系图
```
┌─────────────────┐
│      NP         │
│  ┌───────┐      │
│  │   P   │      │
│  └───────┘      │
│      ● NPC      │
└─────────────────┘

如果 P ≠ NP（未证明）：
P ⊂ NP，NPC 在 NP 中但不在 P 中
```

---

### 26.2 重要的 NPC 问题

#### 1. SAT（布尔可满足性问题）
```
输入：布尔公式 φ
输出：是否存在变量赋值使 φ 为真

Cook-Levin 定理：SAT 是第一个 NPC 问题
```

#### 2. 3-CNF-SAT
```
SAT 的特例：公式为合取范式，每个子句恰有 3 个文字

应用：通过归约证明其他问题是 NPC
```

#### 3. 团问题（Clique）
```
输入：无向图 G=(V,E)，整数 k
输出：是否存在大小为 k 的团（两两相连的顶点子集）

归约：3-CNF-SAT ≤p Clique
```

#### 4. 顶点覆盖（Vertex Cover）
```
输入：无向图 G=(V,E)，整数 k
输出：是否存在大小≤k 的顶点覆盖（每条边至少一个端点在集合中）

归约：Clique ≤p Vertex Cover
```

#### 5. 旅行商问题（TSP）
```
输入：带权完全图，界限 B
输出：是否存在长度≤B 的哈密顿回路

归约：Hamiltonian Cycle ≤p TSP
```

---

### 26.3 证明 NPC 的方法

```
步骤：
1. 证明问题属于 NP（能多项式时间验证）
2. 选择一个已知的 NPC 问题 L
3. 构造从 L 到该问题的多项式时间归约
4. 证明归约的正确性

常用归约链：
SAT → 3-CNF-SAT → Clique → Vertex Cover → Hamiltonian Cycle → TSP
```

## 第 27 章：近似算法（Approximation Algorithms）⭐⭐⭐⭐

### 27.1 为什么需要近似算法？

```
对于 NP-hard 问题：
- 精确解：指数时间，不可行
- 选择：
  1. 只解小规模实例（回溯、分支限界）
  2. 启发式算法（不能保证质量）
  3. 近似算法（保证解的质量）✓
```

### 27.2 顶点覆盖的近似算法

#### 2-近似算法
```javascript
/**
 * 顶点覆盖的 2-近似算法
 * 近似比：2
 */
function vertexCoverApprox(graph) {
    const V = graph.V;
    const covered = new Set();
    const edges = [];
    
    // 收集所有边
    for (let u = 0; u < V; u++) {
        for (let v of graph.getNeighbors(u)) {
            if (u < v.vertex) {
                edges.push([u, v.vertex]);
            }
        }
    }
    
    const uncovered = new Set(edges);
    
    while (uncovered.size > 0) {
        // 任选一条未覆盖的边
        const [u, v] = Array.from(uncovered)[0];
        
        // 加入两个端点
        covered.add(u);
        covered.add(v);
        
        // 移除所有被覆盖的边
        for (let edge of uncovered) {
            if (edge[0] === u || edge[0] === v ||
                edge[1] === u || edge[1] === v) {
                uncovered.delete(edge);
            }
        }
    }
    
    return covered;
}

// 近似比证明：
// 算法选的边集 M 是匹配（无公共顶点）
// 最优解至少要覆盖 M 中的边，所以 |OPT| ≥ |M|
// 算法返回 2|M| 个顶点，所以 |ALG| ≤ 2|OPT|
```

### 27.3 旅行商问题的近似

#### 2-近似（满足三角不等式）
```javascript
/**
 * TSP 的 2-近似算法（满足三角不等式）
 * 1. 求 MST
 * 2. DFS 遍历得到回路
 * 3. 短路跳过重复顶点
 */
function tspApprox(graph) {
    // 1. 求 MST（Prim 算法）
    const { mst } = primMST(graph);
    
    // 2. 构建邻接表
    const mstGraph = new GraphAdjList(graph.V);
    for (let { u, v } of mst) {
        mstGraph.addEdge(u, v);
    }
    
    // 3. DFS 遍历
    const visited = new Array(graph.V).fill(false);
    const tour = [];
    
    function dfs(u) {
        visited[u] = true;
        tour.push(u);
        
        for (let { vertex: v } of mstGraph.getNeighbors(u)) {
            if (!visited[v]) {
                dfs(v);
            }
        }
    }
    
    dfs(0);
    tour.push(0);  // 回到起点
    
    return tour;
}
```

### 27.4 集合覆盖问题

#### 贪心近似算法
```javascript
/**
 * 集合覆盖的贪心近似算法
 * 近似比：H(n) ≈ ln n
 */
function setCoverApprox(universe, sets) {
    const uncovered = new Set(universe);
    const cover = [];
    
    while (uncovered.size > 0) {
        // 选择覆盖最多未覆盖元素的集合
        let bestSet = null;
        let bestCount = 0;
        
        for (let i = 0; i < sets.length; i++) {
            const count = sets[i].filter(x => uncovered.has(x)).length;
            
            if (count > bestCount) {
                bestCount = count;
                bestSet = i;
            }
        }
        
        if (bestSet === null) break;
        
        cover.push(bestSet);
        
        // 从未覆盖中移除
        for (let elem of sets[bestSet]) {
            uncovered.delete(elem);
        }
    }
    
    return cover;
}

// 测试

    [1, 2, 3],

    [6, 7]
];
```
