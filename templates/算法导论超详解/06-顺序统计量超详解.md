# 📚 算法导论 Part III - 顺序统计量超详解

## 第 9 章：找第 k 小的数 - 顺序统计量

### 9.1 什么是顺序统计量？

#### 生活中的例子

**场景 1：考试排名**

```
班级有 30 个学生

你考了第 5 名

问：你的成绩是多少分？

需要找到第 5 小的分数（或第 5 大）

这就是顺序统计量问题！
```

**场景 2：工资中位数**

```
公司有 100 个员工

想知道工资的中位数

中位数 = 第 50 小的工资（排序后中间的数）

这也是顺序统计量问题！
```

**说人话定义：**

```
顺序统计量 = 在一组数中找第 k 小的数

特殊情况：
- 第 1 小：最小值
- 第 n 小：最大值
- 第 n/2 小：中位数

输入：
- 一个数组 A[1..n]
- 一个整数 k（1 ≤ k ≤ n）

输出：
- 数组中第 k 小的元素

例如：
A = [7, 10, 4, 3, 20, 15]
k = 3

排序后：[3, 4, 7, 10, 15, 20]
第 3 小：7
```

### 9.2 方法 1：排序法 - 最简单

#### 核心思想

```
生活比喻：排好序再数

做法：
1. 把数组排序
2. 返回第 k 个元素

简单粗暴！
```

#### 代码实现

```javascript
/**
 * 方法 1：排序法
 * @param {number[]} nums - 输入数组
 * @param {number} k - 第 k 小
 * @returns {number} - 第 k 小的元素
 */
function findKthSmallest_Sort(nums, k) {
    // 排序
    nums.sort((a, b) => a - b);
    
    // 返回第 k 个（索引 k-1）
    return nums[k - 1];
}

// 测试
const nums = [7, 10, 4, 3, 20, 15];
console.log(findKthSmallest_Sort([...nums], 3));
// 输出：7 ✓

console.log(findKthSmallest_Sort([...nums], 1));
// 输出：3（最小值）✓

console.log(findKthSmallest_Sort([...nums], 6));
// 输出：20（最大值）✓
```

#### 时间复杂度

```
排序：O(n log n)
访问：O(1)

总时间：O(n log n)

评价：
✓ 简单易懂
✓ 代码短
✗ 不是最快的
```

### 9.3 方法 2：快速选择算法 - 最快！

#### 核心思想

```
生活比喻：快速排序的分区思想

QuickSelect 算法的做法：
1. 选一个"枢轴"（pivot）
2. 分区：比 pivot 小的放左边，大的放右边
3. pivot 的位置是 i
   - 如果 i == k，找到了！就是 pivot
   - 如果 i > k，在左边继续找
   - 如果 i < k，在右边继续找
4. 重复 1-3，直到找到

关键：
每次都能排除一半的元素！
平均时间 O(n)！
```

#### 代码实现

```javascript
/**
 * 方法 2：快速选择算法
 */
function findKthSmallest_QuickSelect(nums, k) {
    return quickSelect(nums, 0, nums.length - 1, k - 1);
}

/**
 * 快速选择主函数
 */
function quickSelect(arr, left, right, k) {
    if (left === right) {
        return arr[left];
    }
    
    // 分区，得到 pivot 的位置
    const pivotIndex = partition(arr, left, right);
    
    // 判断 pivot 的位置
    if (k === pivotIndex) {
        // 找到了！
        return arr[k];
    } else if (k < pivotIndex) {
        // 在左半部分找
        return quickSelect(arr, left, pivotIndex - 1, k);
    } else {
        // 在右半部分找
        return quickSelect(arr, pivotIndex + 1, right, k);
    }
}

/**
 * 分区操作（和快速排序一样）
 */
function partition(arr, left, right) {
    // 选最右边的元素作为 pivot
    const pivot = arr[right];
    let i = left;  // i 指向第一个大于 pivot 的位置
    
    for (let j = left; j < right; j++) {
        if (arr[j] <= pivot) {
            // 交换 arr[i] 和 arr[j]
            [arr[i], arr[j]] = [arr[j], arr[i]];
            i++;
        }
    }
    
    // 把 pivot 放到中间
    [arr[i], arr[right]] = [arr[right], arr[i]];
    
    return i;  // 返回 pivot 的最终位置
}

// 测试
const nums = [7, 10, 4, 3, 20, 15];
console.log(findKthSmallest_QuickSelect([...nums], 3));
// 输出：7 ✓
```

### 9.4 方法 3：堆方法 - 适合海量数据

#### 核心思想

```
生活比喻：维护一个大小为 k 的最大堆

做法：
1. 创建一个大小为 k 的最大堆
2. 依次读入每个元素
   - 如果堆不满，直接插入
   - 如果堆满了，且当前元素 < 堆顶
     → 弹出堆顶，插入当前元素
3. 最后堆顶就是第 k 小的元素

为什么？
→ 堆里保存的是最小的 k 个数
→ 堆顶是这 k 个数中最大的
→ 所以是第 k 小！
```

#### 代码实现

```javascript
/**
 * 方法 3：最大堆方法
 */
function findKthSmallest_Heap(nums, k) {
    // 创建最大堆（用负数模拟）
    const maxHeap = [];
    
    for (let num of nums) {
        if (maxHeap.length < k) {
            // 堆不满，直接插入
            maxHeap.push(-num);
            heapifyUp(maxHeap, maxHeap.length - 1);
        } else if (-num > maxHeap[0]) {
            // 当前元素更小，替换堆顶
            maxHeap[0] = -num;
            heapifyDown(maxHeap, 0, k);
        }
    }
    
    // 堆顶是第 k 小（取反）
    return -maxHeap[0];
}

/**
 * 上浮调整
 */
function heapifyUp(heap, i) {
    while (i > 0) {
        const parent = Math.floor((i - 1) / 2);
        if (heap[i] <= heap[parent]) break;
        [heap[i], heap[parent]] = [heap[parent], heap[i]];
        i = parent;
    }
}

/**
 * 下沉调整
 */
function heapifyDown(heap, i, size) {
    while (true) {
        let largest = i;
        const left = 2 * i + 1;
        const right = 2 * i + 2;
        
        if (left < size && heap[left] > heap[largest]) {
            largest = left;
        }
        if (right < size && heap[right] > heap[largest]) {
            largest = right;
        }
        
        if (largest === i) break;
        [heap[i], heap[largest]] = [heap[largest], heap[i]];
        i = largest;
    }
}

// 测试
const nums = [7, 10, 4, 3, 20, 15];
console.log(findKthSmallest_Heap(nums, 3));
// 输出：7 ✓
```

#### 时间复杂度

```
建堆：O(k)
处理 n 个元素：O((n-k) log k)

总时间：O(n log k)

优点：
✓ 适合海量数据（流式处理）
✓ 不需要一次性加载所有数据
✓ k 很小时很快

缺点：
✗ k 接近 n 时慢
```

```javascript
function topKLargest(nums, k) {
```

```javascript
/**
 * 找中位数
 */
function findMedian(nums) {
    const n = nums.length;
    
    if (n % 2 === 1) {
        // 奇数个，找第 n/2+1 小
        return findKthSmallest_QuickSelect([...nums], Math.floor(n / 2) + 1);
    } else {
        // 偶数个，找中间两个的平均
        const m1 = findKthSmallest_QuickSelect([...nums], n / 2);
        const m2 = findKthSmallest_QuickSelect([...nums], n / 2 + 1);
        return (m1 + m2) / 2;
    }
}

// 测试
console.log(findMedian([7, 10, 4, 3, 20]));
// 输出：7 ✓（排序后 [3,4,7,10,20]，中间是 7）
```

### 9.7 完整代码模板（背诵版）

```javascript
/**
 * 快速选择算法 - 找第 k 小
 */
function findKthSmallest(nums, k) {
    return quickSelect(nums, 0, nums.length - 1, k - 1);
}

function quickSelect(arr, left, right, k) {
    if (left === right) return arr[left];
    
    const pivotIndex = partition(arr, left, right);
    
    if (k === pivotIndex) {
        return arr[k];
    } else if (k < pivotIndex) {
        return quickSelect(arr, left, pivotIndex - 1, k);
    } else {
        return quickSelect(arr, pivotIndex + 1, right, k);
    }
}

function partition(arr, left, right) {
    const pivot = arr[right];
    let i = left;
    
    for (let j = left; j < right; j++) {
        if (arr[j] <= pivot) {
            [arr[i], arr[j]] = [arr[j], arr[i]];
            i++;
        }
    }
    
    [arr[i], arr[right]] = [arr[right], arr[i]];
    return i;
}

/**
 * 堆方法 - 第 k 小
 */
function findKthSmallestHeap(nums, k) {

            maxHeap[0] = -num;
            heapifyDown(maxHeap, 0, k);
        }
    }
    
    return -maxHeap[0];
}

}

        [heap[i], heap[largest]] = [heap[largest], heap[i]];
        i = largest;

}
```
