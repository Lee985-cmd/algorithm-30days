# 📚 算法导论第 33 章 - NP 完全性证明技巧

## 33.1 证明方法概述

```
证明一个问题 Q 是 NP 完全的步骤：

1. 证明 Q ∈ NP
   - 给出一个多项式时间的验证算法

2. 选择一个已知的 NP 完全问题 Q'
   - 通常选择 SAT、3-CNF-SAT、CLIQUE、VERTEX-COVER 等

3. 构造从 Q' 到 Q 的多项式时间归约 f
   - 对任意输入 x，x ∈ Q' ⟺ f(x) ∈ Q

4. 证明归约的正确性
   - 证明 f 是多项式时间的
   - 证明 x ∈ Q' ⟺ f(x) ∈ Q

5. 结论
   - Q 是 NP 完全的
```

## 33.2 SAT 问题（布尔可满足性）

```javascript
/**
 * SAT 问题示例
 * 
 * 问题：给定一个布尔公式，是否存在一组变量赋值使得公式为真？
 * 
 * Cook-Levin 定理：SAT 是 NP 完全的
 */

// 3-SAT 问题示例（每个子句恰好 3 个文字）
// (x₁ ∨ x₂ ∨ ¬x₃) ∧ (¬x₁ ∨ x₃ ∨ x₄) ∧ (x₂ ∨ ¬x₃ ∨ ¬x₄)

function solve3SAT(clauses, numVars) {
    // 暴力搜索所有可能的赋值（指数时间）
    for (let i = 0; i < Math.pow(2, numVars); i++) {
        const assignment = [];
        
        // 生成当前赋值
        for (let j = 0; j < numVars; j++) {
            assignment.push((i >> j) & 1);
        }
        
        // 检查是否满足所有子句
        if (satisfiesAllClauses(clauses, assignment)) {
            return { satisfiable: true, assignment };
        }
    }
    
    return { satisfiable: false, assignment: null };
}

function satisfiesAllClauses(clauses, assignment) {
    for (const clause of clauses) {
        if (!satisfiesClause(clause, assignment)) {
            return false;
        }
    }
    return true;
}

function satisfiesClause(clause, assignment) {
    for (const literal of clause) {
        const varIndex = Math.abs(literal) - 1;
        const value = assignment[varIndex];
        
        // 如果是正文字且值为 1，或负文字且值为 0，则子句为真
        if ((literal > 0 && value === 1) || (literal < 0 && value === 0)) {
            return true;
        }
    }
    return false;
}

// 测试
const clauses = [
    [1, 2, -3],    // (x₁ ∨ x₂ ∨ ¬x₃)
    [-1, 3, 4],    // (¬x₁ ∨ x₃ ∨ x₄)
    [2, -3, -4]    // (x₂ ∨ ¬x₃ ∨ ¬x₄)
];

const result = solve3SAT(clauses, 4);
console.log('3-SAT 可满足吗？', result.satisfiable);
if (result.satisfiable) {
    console.log('赋值:', result.assignment);
}
```

## 33.3 团问题（CLIQUE）

```javascript
/**
 * 团问题：图中是否存在大小为 k 的完全子图？
 * 
 * 从 3-SAT 归约到 CLIQUE
 */
function hasClique(graph, k) {
    const n = graph.length;
    
    // 暴力搜索所有大小为 k 的顶点子集
    const subsets = generateSubsets(n, k);
    
    for (const subset of subsets) {
        if (isClique(graph, subset)) {
            return { exists: true, clique: subset };
        }
    }
    
    return { exists: false, clique: null };
}

function generateSubsets(n, k) {
    const result = [];
    const subset = [];
    
    function backtrack(start, remaining) {
        if (remaining === 0) {
            result.push([...subset]);
            return;
        }
        
        for (let i = start; i < n; i++) {
            subset.push(i);
            backtrack(i + 1, remaining - 1);
            subset.pop();
        }
    }
    
    backtrack(0, k);
    return result;
}

function isClique(graph, subset) {
    for (let i = 0; i < subset.length; i++) {
        for (let j = i + 1; j < subset.length; j++) {
            if (!graph[subset[i]][subset[j]]) {
                return false;
            }
        }
    }
    return true;
}

// 测试
const graph = [
    [false, true, true, false],
    [true, false, true, true],
    [true, true, false, true],
    [false, true, true, false]
];

const result = hasClique(graph, 3);
console.log('存在大小为 3 的团吗？', result.exists);
if (result.exists) {
    console.log('团的顶点:', result.clique);
}
```

## 33.4 顶点覆盖问题（VERTEX-COVER）

```javascript
/**
 * 顶点覆盖问题：是否存在大小不超过 k 的顶点集合，
 * 覆盖图中的所有边？
 * 
 * 从 3-SAT 归约到 VERTEX-COVER
 */
function hasVertexCover(graph, k) {
    const n = graph.vertices;
    const edges = graph.edges;
    
    // 暴力搜索
    for (let size = 0; size <= k; size++) {
        const subsets = generateSubsets(n, size);
        
        for (const subset of subsets) {
            if (coversAllEdges(subset, edges)) {
                return { exists: true, cover: subset };
            }
        }
    }
    
    return { exists: false, cover: null };
}

function coversAllEdges(vertices, edges) {
    const vertexSet = new Set(vertices);
    
    for (const edge of edges) {
        if (!vertexSet.has(edge.u) && !vertexSet.has(edge.v)) {
            return false;
        }
    }
    
    return true;
}

// 测试
const graph = {
    vertices: 4,
    edges: [
        { u: 0, v: 1 },
        { u: 0, v: 2 },
        { u: 1, v: 2 },
        { u: 1, v: 3 },
        { u: 2, v: 3 }
    ]
};

const result = hasVertexCover(graph, 2);
console.log('存在大小不超过 2 的顶点覆盖吗？', result.exists);
if (result.exists) {
    console.log('顶点覆盖:', result.cover);
}
```

## 33.6 近似算法设计

```javascript
/**
 * 最大可满足性问题（MAX-3-SAT）的随机近似算法
 * 期望近似比：7/8
 */
function randomMax3SAT(clauses, numVars) {
    const assignment = [];
    
    // 随机赋值
    for (let i = 0; i < numVars; i++) {
        assignment.push(Math.random() < 0.5 ? 0 : 1);
    }
    
    // 计算满足的子句数
    let satisfied = 0;
    for (const clause of clauses) {
        if (satisfiesClause(clause, assignment)) {
            satisfied++;
        }
    }
    
    return { assignment, satisfiedCount: satisfied };
}

// 这是一个 7/8-近似算法
// 期望满足至少 7/8 的最优子句数
```
