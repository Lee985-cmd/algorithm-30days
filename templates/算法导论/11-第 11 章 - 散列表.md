# 📚 算法导论第 11 章 - 散列表

## 11.1 直接寻址表

```
适用条件：
- 关键字范围很小（0 到 k）
- k = O(n)

时间复杂度：O(1)
空间复杂度：O(k)
```

## 11.2 散列表

### 核心思想

```
通过散列函数 h 将关键字映射到槽位：
h: U → {0, 1, ..., m-1}

其中：
- U 是关键字全域
- m 是槽位数
```

## 11.3 链接法解决冲突

```javascript
/**
 * 散列表 - 链接法
 */
class HashTableChaining {
    constructor(size = 16) {
        this.size = size;
        this.table = new Array(size).fill(null).map(() => []);
    }
    
    hash(key) {
        // 简单散列函数
        let hash = 0;
        for (let i = 0; i < key.length; i++) {
            hash += key.charCodeAt(i);
        }
        return hash % this.size;
    }
    
    set(key, value) {
        const index = this.hash(key);
        
        // 检查是否已存在
        for (let item of this.table[index]) {
            if (item.key === key) {
                item.value = value;
                return;
            }
        }
        
        // 添加新键值对
        this.table[index].push({ key, value });
    }
    
    get(key) {
        const index = this.hash(key);
        
        for (let item of this.table[index]) {
            if (item.key === key) {
                return item.value;
            }
        }
        
        return null;
    }
    
    remove(key) {
        const index = this.hash(key);
        const chain = this.table[index];
        
        for (let i = 0; i < chain.length; i++) {
            if (chain[i].key === key) {
                chain.splice(i, 1);
                return true;
            }
        }
        
        return false;
    }
}

// 测试
const hashTable = new HashTableChaining();
hashTable.set('name', 'Alice');
hashTable.set('age', 25);
console.log(hashTable.get('name')); // Alice
```

## 11.4 开放寻址法

### 线性探测

```javascript
/**
 * 散列表 - 开放寻址法（线性探测）
 */
class HashTableOpenAddressing {
    constructor(size = 31) {
        this.size = size;
        this.table = new Array(size).fill(null);
        this.count = 0;
    }
    
    hash(key) {
        let hash = 0;
        for (let i = 0; i < key.length; i++) {
            hash += key.charCodeAt(i);
        }
        return hash % this.size;
    }
    
    probe(key, i) {
        // 线性探测
        return (this.hash(key) + i) % this.size;
    }
    
    set(key, value) {
        if (this.count >= this.size / 2) {
            this.rehash();
        }
        
        let i = 0;
        while (i < this.size) {
            const index = this.probe(key, i);
            
            if (this.table[index] === null || this.table[index].deleted) {
                this.table[index] = { key, value };
                this.count++;
                return;
            }
            
            if (this.table[index].key === key) {
                this.table[index].value = value;
                return;
            }
            
            i++;
        }
        
        throw new Error('Hash table is full');
    }
    
    get(key) {
        let i = 0;
        while (i < this.size) {
            const index = this.probe(key, i);
            
            if (this.table[index] === null) {
                return null;
            }
            
            if (!this.table[index].deleted && this.table[index].key === key) {
                return this.table[index].value;
            }
            
            i++;
        }
        
        return null;
    }
    
    remove(key) {
        let i = 0;
        while (i < this.size) {
            const index = this.probe(key, i);
            
            if (this.table[index] === null) {
                return false;
            }
            
            if (!this.table[index].deleted && this.table[index].key === key) {
                this.table[index].deleted = true;
                this.count--;
                return true;
            }
            
            i++;
        }
        
        return false;
    }
    
    rehash() {
        const oldTable = [...this.table];
        this.size *= 2;
        this.table = new Array(this.size).fill(null);
        this.count = 0;
        
        for (let item of oldTable) {
            if (item && !item.deleted) {
                this.set(item.key, item.value);
            }
        }
    }
}

// 测试
const hash = new HashTableOpenAddressing();
hash.set('name', 'Bob');
hash.set('age', 30);
console.log(hash.get('name')); // Bob
```
