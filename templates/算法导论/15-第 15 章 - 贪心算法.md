## 15.1 活动选择问题

```
问题描述：
有 n 个活动，每个活动有开始时间 sᵢ 和结束时间 fᵢ
同一时间只能参加一个活动
求最多能参加多少个活动
```

### 贪心解法

```javascript
/**
 * 活动选择问题 - 贪心算法
 */
function activitySelection(activities) {
    // 按结束时间排序
    activities.sort((a, b) => a.finish - b.finish);
    
    const selected = [];
    let lastFinish = 0;
    
    for (let activity of activities) {
        if (activity.start >= lastFinish) {
            selected.push(activity);
            lastFinish = activity.finish;
        }
    }
    
    return selected;
}

// 测试
const activities = [
    { start: 1, finish: 4 },
    { start: 3, finish: 5 },
    { start: 0, finish: 6 },
    { start: 5, finish: 7 },
    { start: 3, finish: 9 },
    { start: 5, finish: 9 },
    { start: 6, finish: 10 },
    { start: 8, finish: 11 },
    { start: 8, finish: 12 },
    { start: 2, finish: 14 },
    { start: 12, finish: 16 }
];

const result = activitySelection(activities);
console.log('最多参加活动数:', result.length); // 4 ✓
console.log('活动列表:', result);
// [{1,4}, {5,7}, {8,11}, {12,16}] ✓
```

## 15.2 霍夫曼编码

```javascript
/**
 * 霍夫曼编码
 */
class HuffmanNode {
    constructor(char, freq) {
        this.char = char;
        this.freq = freq;
        this.left = null;
        this.right = null;
    }
}

class MinHeap {
    constructor() {
        this.heap = [];
    }
    
    insert(node) {
        this.heap.push(node);
        this.bubbleUp(this.heap.length - 1);
    }
    
    extractMin() {
        if (this.heap.length === 0) return null;
        
        const min = this.heap[0];
        const last = this.heap.pop();
        
        if (this.heap.length > 0) {
            this.heap[0] = last;
            this.sinkDown(0);
        }
        
        return min;
    }
    
    bubbleUp(index) {
        while (index > 0) {
            const parent = Math.floor((index - 1) / 2);
            if (this.heap[parent].freq <= this.heap[index].freq) break;
            
            [this.heap[parent], this.heap[index]] = [this.heap[index], this.heap[parent]];
            index = parent;
        }
    }
    
    sinkDown(index) {
        const length = this.heap.length;
        
        while (true) {
            const left = 2 * index + 1;
            const right = 2 * index + 2;
            let smallest = index;
            
            if (left < length && this.heap[left].freq < this.heap[smallest].freq) {
                smallest = left;
            }
            
            if (right < length && this.heap[right].freq < this.heap[smallest].freq) {
                smallest = right;
            }
            
            if (smallest === index) break;
            
            [this.heap[smallest], this.heap[index]] = [this.heap[index], this.heap[smallest]];
            index = smallest;
        }
    }
    
    size() {
        return this.heap.length;
    }
}

function huffmanEncoding(chars, freqs) {
    const n = chars.length;
    const heap = new MinHeap();
    
    // 创建叶子节点
    for (let i = 0; i < n; i++) {
        heap.insert(new HuffmanNode(chars[i], freqs[i]));
    }
    
    // 构建霍夫曼树
    for (let i = 1; i < n; i++) {
        const left = heap.extractMin();
        const right = heap.extractMin();
        
        const merged = new HuffmanNode(null, left.freq + right.freq);
        merged.left = left;
        merged.right = right;
        
        heap.insert(merged);
    }
    
    return heap.extractMin();
}

function generateCodes(root, prefix = '', codes = {}) {
    if (root === null) return;
    
    if (root.char !== null) {
        codes[root.char] = prefix;
        return;
    }
    
    generateCodes(root.left, prefix + '0', codes);
    generateCodes(root.right, prefix + '1', codes);
    
    return codes;
}

// 测试
const chars = ['a', 'b', 'c', 'd', 'e', 'f'];
const freqs = [5, 9, 12, 13, 16, 45];

const root = huffmanEncoding(chars, freqs);
const codes = generateCodes(root);

console.log('霍夫曼编码:', codes);
// { a: '1100', b: '1101', c: '100', d: '101', e: '111', f: '0' } ✓
```
