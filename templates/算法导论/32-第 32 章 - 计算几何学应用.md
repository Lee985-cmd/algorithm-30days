## 32.1 多边形三角剖分

```javascript
/**
 * 简单多边形的三角剖分（耳切法）
 */
function triangulatePolygon(polygon) {
    const n = polygon.length;
    if (n < 3) return [];
    
    const triangles = [];
    const vertices = [...polygon];
    
    while (vertices.length > 3) {
        // 找一个"耳朵"（凸顶点，其相邻两顶点连线在多边形内）
        for (let i = 0; i < vertices.length; i++) {
            const prev = vertices[(i - 1 + vertices.length) % vertices.length];
            const curr = vertices[i];
            const next = vertices[(i + 1) % vertices.length];
            
            if (isEar(prev, curr, next, vertices)) {
                triangles.push([prev, curr, next]);
                vertices.splice(i, 1);
                break;
            }
        }
    }
    
    // 最后一个三角形
    if (vertices.length === 3) {
        triangles.push(vertices);
    }
    
    return triangles;
}

function isEar(prev, curr, next, polygon) {
    // 检查角是否为凸角
    const cross = (curr.x - prev.x) * (next.y - prev.y) - 
                  (curr.y - prev.y) * (next.x - prev.x);
    
    if (cross <= 0) return false;
    
    // 检查三角形内是否包含其他顶点
    for (const point of polygon) {
        if (point === prev || point === curr || point === next) continue;
        
        if (pointInTriangle(point, prev, curr, next)) {
            return false;
        }
    }
    
    return true;
}

function pointInTriangle(point, a, b, c) {
    const crossSign = (p1, p2, p3) => {
        return (p1.x - p3.x) * (p2.y - p3.y) - (p2.x - p3.x) * (p1.y - p3.y);
    };
    
    const d1 = crossSign(point, a, b);
    const d2 = crossSign(point, b, c);
    const d3 = crossSign(point, c, a);
    
    const hasNeg = (d1 < 0) || (d2 < 0) || (d3 < 0);
    const hasPos = (d1 > 0) || (d2 > 0) || (d3 > 0);
    
    return !(hasNeg && hasPos);
}

// 测试
const polygon = [
    { x: 0, y: 0 },
    { x: 4, y: 0 },
    { x: 4, y: 3 },
    { x: 2, y: 5 },
    { x: 0, y: 3 }
];

const triangles = triangulatePolygon(polygon);
console.log('三角剖分结果:', triangles);
```

## 32.2 线段扫描算法

```javascript
/**
 * 扫描线算法检测线段相交
 * 时间复杂度：O(n log n)
 */
function findIntersections(segments) {
    const events = [];
    
    // 创建事件点
    for (const seg of segments) {
        const left = seg.p1.x < seg.p2.x ? seg.p1 : seg.p2;
        const right = seg.p1.x < seg.p2.x ? seg.p2 : seg.p1;
        
        events.push({ type: 'LEFT', point: left, segment: seg });
        events.push({ type: 'RIGHT', point: right, segment: seg });
    }
    
    // 按 x 坐标排序事件点
    events.sort((a, b) => a.point.x - b.point.x);
    
    const sweepLine = [];
    const intersections = [];
    
    for (const event of events) {
        if (event.type === 'LEFT') {
            // 插入扫描线
            sweepLine.push(event.segment);
            
            // 检查与相邻线段的相交
            const idx = sweepLine.indexOf(event.segment);
            if (idx > 0 && segmentsIntersect(
                sweepLine[idx - 1].p1, sweepLine[idx - 1].p2,
                event.segment.p1, event.segment.p2)) {
                intersections.push([sweepLine[idx - 1], event.segment]);
            }
            if (idx < sweepLine.length - 1 && segmentsIntersect(
                event.segment.p1, event.segment.p2,
                sweepLine[idx + 1].p1, sweepLine[idx + 1].p2)) {
                intersections.push([event.segment, sweepLine[idx + 1]]);
            }
        } else {
            // 从扫描线移除
            const idx = sweepLine.indexOf(event.segment);
            if (idx >= 0) {
                sweepLine.splice(idx, 1);
            }
        }
    }
    
    return intersections;
}

// 测试
const segments = [
    { p1: { x: 0, y: 0 }, p2: { x: 4, y: 4 } },
    { p1: { x: 0, y: 4 }, p2: { x: 4, y: 0 } },
    { p1: { x: 1, y: 0 }, p2: { x: 1, y: 4 } }
];

const result = findIntersections(segments);
console.log('相交线段对:', result.length);
```
