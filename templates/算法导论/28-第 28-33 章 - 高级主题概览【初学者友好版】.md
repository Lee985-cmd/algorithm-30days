### Strassen 矩阵乘法

```javascript
/**
 * Strassen 算法
 * 
 * 传统矩阵乘法：O(n³)
 * Strassen: O(n^log₂7) ≈ O(n^2.81)
 */

// 传统方法
function multiplyMatrix(A, B) {
    const n = A.length;
    const C = Array.from({ length: n }, () => new Array(n).fill(0));
    
    for (let i = 0; i < n; i++) {
        for (let j = 0; j < n; j++) {
            for (let k = 0; k < n; k++) {
                C[i][j] += A[i][k] * B[k][j];
            }
        }
    }
    
    return C;
}

// ⏱️ O(n³)
```

// 扩展欧几里得算法
function extendedGCD(a, b) {
    if (b === 0) {
        return { g: a, x: 1, y: 0 };
    }
    const { g, x, y } = extendedGCD(b, a % b);
    return {
        g,
        x: y,
        y: x - Math.floor(a / b) * y
    };
}

// RSA 加密基础
// 用于网络安全、数字货币等
