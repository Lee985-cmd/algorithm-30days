# 📚 算法导论第 25 章 - 字符串匹配

## 25.1 KMP 算法

```javascript
/**
 * KMP 算法 - 字符串匹配
 */
function kmpSearch(text, pattern) {
    const n = text.length;
    const m = pattern.length;
    
    // 构建部分匹配表（前缀函数）
    const pi = computePrefixFunction(pattern);
    
    let q = 0;  // 已匹配的字符数
    
    for (let i = 0; i < n; i++) {
        while (q > 0 && pattern[q] !== text[i]) {
            q = pi[q - 1];
        }
        
        if (pattern[q] === text[i]) {
            q++;
        }
        
        if (q === m) {
            console.log(`Pattern occurs with shift ${i - m + 1}`);
            q = pi[q - 1];
        }
    }
}

function computePrefixFunction(pattern) {
    const m = pattern.length;
    const pi = new Array(m).fill(0);
    
    let k = 0;
    
    for (let q = 1; q < m; q++) {
        while (k > 0 && pattern[k] !== pattern[q]) {
            k = pi[k - 1];
        }
        
        if (pattern[k] === pattern[q]) {
            k++;
        }
        
        pi[q] = k;
    }
    
    return pi;
}

// 测试
const text = "ababcabcacbab";
const pattern = "abcac";
kmpSearch(text, pattern);
// Pattern occurs with shift 4 ✓
```
