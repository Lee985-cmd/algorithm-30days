## 22.1 Dijkstra 算法

```javascript
/**
 * Dijkstra 算法 - 单源最短路径（无负权边）
 */
function dijkstra(graph, weights, source) {
    const n = graph.vertices;
    const dist = new Array(n).fill(Infinity);
    const parent = new Array(n).fill(null);
    const visited = new Array(n).fill(false);
    
    dist[source] = 0;
    
    for (let count = 0; count < n; count++) {
        // 找最小距离的顶点
        let u = -1;
        let minDist = Infinity;
        
        for (let v = 0; v < n; v++) {
            if (!visited[v] && dist[v] < minDist) {
                minDist = dist[v];
                u = v;
            }
        }
        
        if (u === -1) break;
        visited[u] = true;
        
        // 更新相邻顶点
        for (const v of graph.adj[u]) {
            if (!visited[v]) {
                const newDist = dist[u] + weights[`${u}-${v}`];
                if (newDist < dist[v]) {
                    dist[v] = newDist;
                    parent[v] = u;
                }
            }
        }
    }
    
    return { dist, parent };
}

// 测试
const g = new Graph(5);
g.addEdge(0, 1);
g.addEdge(0, 2);
g.addEdge(1, 2);
g.addEdge(1, 3);
g.addEdge(2, 3);
g.addEdge(3, 4);

const weights = {
    '0-1': 10, '0-2': 5,
    '1-2': 2, '1-3': 1,
    '2-3': 2,
    '3-4': 7
};

const result = dijkstra(g, weights, 0);
console.log('Dijkstra 最短路径:', result.dist);
// [0, 10, 5, 7, 14] ✓
```

## 22.2 Bellman-Ford 算法

```javascript
/**
 * Bellman-Ford 算法 - 可处理负权边
 */
function bellmanFord(vertices, edges, source) {
    const dist = new Array(vertices).fill(Infinity);
    const parent = new Array(vertices).fill(null);
    
    dist[source] = 0;
    
    // 松弛操作，执行 V-1 次
    for (let i = 0; i < vertices - 1; i++) {
        for (const edge of edges) {
            const { u, v, weight } = edge;
            
            if (dist[u] !== Infinity && dist[u] + weight < dist[v]) {
                dist[v] = dist[u] + weight;
                parent[v] = u;
            }
        }
    }
    
    // 检查负权环
    for (const edge of edges) {
        const { u, v, weight } = edge;
        if (dist[u] !== Infinity && dist[u] + weight < dist[v]) {
            return { hasNegativeCycle: true };
        }
    }
    
    return { dist, parent, hasNegativeCycle: false };
}

// 测试
const edges = [
    { u: 0, v: 1, weight: 6 },
    { u: 0, v: 3, weight: 7 },
    { u: 1, v: 2, weight: 5 },
    { u: 1, v: 3, weight: 8 },
    { u: 1, v: 4, weight: -4 },
    { u: 2, v: 1, weight: -2 },
    { u: 3, v: 2, weight: -3 },
    { u: 3, v: 4, weight: 9 },
    { u: 4, v: 0, weight: 2 },
    { u: 4, v: 2, weight: 7 }
];

const result = bellmanFord(5, edges, 0);
console.log('Bellman-Ford 最短路径:', result.dist);
```
