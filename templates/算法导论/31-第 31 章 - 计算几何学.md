## 31.1 线段相交检测

```javascript
/**
 * 向量叉积
 */
function crossProduct(p0, p1, p2) {
    return (p1.x - p0.x) * (p2.y - p0.y) - 
           (p1.y - p0.y) * (p2.x - p0.x);
}

/**
 * 判断点是否在线段上
 */
function onSegment(p, p1, p2) {
    return p.x >= Math.min(p1.x, p2.x) && p.x <= Math.max(p1.x, p2.x) &&
           p.y >= Math.min(p1.y, p2.y) && p.y <= Math.max(p1.y, p2.y);
}

/**
 * 判断两线段是否相交
 */
function segmentsIntersect(p1, p2, p3, p4) {
    const d1 = crossProduct(p3, p4, p1);
    const d2 = crossProduct(p3, p4, p2);
    const d3 = crossProduct(p1, p2, p3);
    const d4 = crossProduct(p1, p2, p4);
    
    // 一般情况：端点在另一线段的两侧
    if (((d1 > 0 && d2 < 0) || (d1 < 0 && d2 > 0)) &&
        ((d3 > 0 && d4 < 0) || (d3 < 0 && d4 > 0))) {
        return true;
    }
    
    // 特殊情况：端点在线段上
    if (d1 === 0 && onSegment(p1, p3, p4)) return true;
    if (d2 === 0 && onSegment(p2, p3, p4)) return true;
    if (d3 === 0 && onSegment(p3, p1, p2)) return true;
    if (d4 === 0 && onSegment(p4, p1, p2)) return true;
    
    return false;
}

// 测试
const p1 = { x: 0, y: 0 };
const p2 = { x: 4, y: 4 };
const p3 = { x: 0, y: 4 };
const p4 = { x: 4, y: 0 };

console.log('线段相交吗？', segmentsIntersect(p1, p2, p3, p4)); // true ✓
```

## 31.2 凸包问题

### Graham 扫描法

```javascript
/**
 * Graham 扫描法求凸包
 */
function grahamScan(points) {
    if (points.length < 3) return points;
    
    // 找最下最左的点
    let start = points[0];
    let startIndex = 0;
    
    for (let i = 1; i < points.length; i++) {
        if (points[i].y < start.y || 
            (points[i].y === start.y && points[i].x < start.x)) {
            start = points[i];
            startIndex = i;
        }
    }
    
    // 按极角排序
    points.splice(startIndex, 1);
    points.sort((a, b) => {
        const angleA = Math.atan2(a.y - start.y, a.x - start.x);
        const angleB = Math.atan2(b.y - start.y, b.x - start.x);
        return angleA - angleB;
    });
    
    // Graham 扫描
    const stack = [start];
    
    for (const point of points) {
        while (stack.length >= 2) {
            const top = stack[stack.length - 1];
            const nextToTop = stack[stack.length - 2];
            
            if (crossProduct(nextToTop, top, point) <= 0) {
                stack.pop();
            } else {
                break;
            }
        }
        
        stack.push(point);
    }
    
    return stack;
}

// 测试
const points = [
    { x: 0, y: 3 },
    { x: 1, y: 1 },
    { x: 2, y: 2 },
    { x: 4, y: 4 },
    { x: 0, y: 0 },
    { x: 3, y: 0 }
];

const convexHull = grahamScan(points);
console.log('凸包顶点:', convexHull);
```

## 31.3 最近点对问题

```javascript
/**
 * 分治法求最近点对
 */
function closestPair(points) {
    if (points.length < 2) return { distance: Infinity, pair: null };
    
    const sortedX = [...points].sort((a, b) => a.x - b.x);
    const sortedY = [...points].sort((a, b) => a.y - b.y);
    
    return closestPairRecursive(sortedX, sortedY);
}

function closestPairRecursive(X, Y) {
    const n = X.length;
    
    if (n <= 3) {
        return bruteForce(X);
    }
    
    const mid = Math.floor(n / 2);
    const midPoint = X[mid];
    
    const XL = X.slice(0, mid);
    const XR = X.slice(mid);
    
    const midY = new Set(XL.map(p => `${p.x},${p.y}`));
    const YL = Y.filter(p => midY.has(`${p.x},${p.y}`));
    const YR = Y.filter(p => !midY.has(`${p.x},${p.y}`));
    
    const leftResult = closestPairRecursive(XL, YL);
    const rightResult = closestPairRecursive(XR, YR);
    
    let delta = Math.min(leftResult.distance, rightResult.distance);
    let result = leftResult.distance < rightResult.distance ? leftResult : rightResult;
    
    // 检查跨越中线的点对
    const strip = Y.filter(p => Math.abs(p.x - midPoint.x) < delta);
    
    for (let i = 0; i < strip.length; i++) {
        for (let j = i + 1; j < strip.length && strip[j].y - strip[i].y < delta; j++) {
            const dist = distance(strip[i], strip[j]);
            if (dist < delta) {
                delta = dist;
                result = { distance: dist, pair: [strip[i], strip[j]] };
            }
        }
    }
    
    return result;
}

function bruteForce(points) {
    let minDist = Infinity;
    let pair = null;
    
    for (let i = 0; i < points.length; i++) {
        for (let j = i + 1; j < points.length; j++) {
            const dist = distance(points[i], points[j]);
            if (dist < minDist) {
                minDist = dist;
                pair = [points[i], points[j]];
            }
        }
    }
    
    return { distance: minDist, pair };
}

function distance(p1, p2) {
    return Math.sqrt((p1.x - p2.x) ** 2 + (p1.y - p2.y) ** 2);
}

// 测试
const points = [
    { x: 2, y: 3 },
    { x: 12, y: 30 },
    { x: 40, y: 50 },
    { x: 5, y: 12 },
    { x: 30, y: 20 },
    { x: 3, y: 4 }
];

const result = closestPair(points);
console.log('最近点对距离:', result.distance.toFixed(2));
console.log('点对:', result.pair);
```
