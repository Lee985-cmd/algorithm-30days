## 24.1 Ford-Fulkerson 方法

```javascript
/**
 * Ford-Fulkerson 算法 - 最大流
 */
function fordFulkerson(capacity, source, sink) {
    const n = capacity.length;
    const residual = capacity.map(row => [...row]);
    const parent = new Array(n);
    
    let maxFlow = 0;
    
    // BFS 找增广路径
    function bfs() {
        parent.fill(-1);
        parent[source] = null;
        
        const queue = [source];
        
        while (queue.length > 0) {
            const u = queue.shift();
            
            for (let v = 0; v < n; v++) {
                if (parent[v] === -1 && residual[u][v] > 0) {
                    parent[v] = u;
                    if (v === sink) return true;
                    queue.push(v);
                }
            }
        }
        
        return false;
    }
    
    // 不断找增广路径
    while (bfs()) {
        // 找路径上的最小剩余容量
        let pathFlow = Infinity;
        let v = sink;
        
        while (parent[v] !== null) {
            const u = parent[v];
            pathFlow = Math.min(pathFlow, residual[u][v]);
            v = u;
        }
        
        // 更新残余网络
        v = sink;
        while (parent[v] !== null) {
            const u = parent[v];
            residual[u][v] -= pathFlow;
            residual[v][u] += pathFlow;
            v = u;
        }
        
        maxFlow += pathFlow;
    }
    
    return maxFlow;
}

// 测试
const capacity = [
    [0, 16, 13, 0, 0, 0],
    [0, 0, 10, 12, 0, 0],
    [0, 4, 0, 0, 14, 0],
    [0, 0, 9, 0, 0, 20],
    [0, 0, 0, 7, 0, 4],
    [0, 0, 0, 0, 0, 0]
];

const maxFlow = fordFulkerson(capacity, 0, 5);
console.log('最大流:', maxFlow); // 23 ✓
```
