# 📚 算法导论第 14 章 - 动态规划

## 14.1 钢条切割问题

```
问题描述：
给定一根长度为 n 的钢条和价格表 p[i]（长度为 i 的钢条价格）
如何切割使得总收益最大？
```

### 解法

```javascript
/**
 * 动态规划 - 钢条切割
 */

// 自顶向下递归（指数时间，不好）
function cutRodRecursive(p, n) {
    if (n === 0) return 0;
    
    let maxVal = -Infinity;
    for (let i = 1; i <= n; i++) {
        maxVal = Math.max(maxVal, p[i] + cutRodRecursive(p, n - i));
    }
    
    return maxVal;
}

// 带备忘录的自顶向下（O(n²)，好）
function memoizedCutRod(p, n) {
    const memo = new Array(n + 1).fill(-Infinity);
    return helper(p, n, memo);
}

function helper(p, n, memo) {
    if (memo[n] >= 0) return memo[n];
    if (n === 0) return 0;
    
    let maxVal = -Infinity;
    for (let i = 1; i <= n; i++) {
        maxVal = Math.max(maxVal, p[i] + helper(p, n - i, memo));
    }
    
    memo[n] = maxVal;
    return maxVal;
}

// 自底向上（O(n²)，推荐）
function bottomUpCutRod(p, n) {
    const dp = new Array(n + 1).fill(0);
    
    for (let j = 1; j <= n; j++) {
        let maxVal = -Infinity;
        for (let i = 1; i <= j; i++) {
            maxVal = Math.max(maxVal, p[i] + dp[j - i]);
        }
        dp[j] = maxVal;
    }
    
    return dp[n];
}

// 测试
const prices = [0, 1, 5, 8, 9, 10, 17, 17, 20, 24, 30];
console.log('最大收益:', bottomUpCutRod(prices, 4)); // 10 ✓
console.log('最大收益:', bottomUpCutRod(prices, 7)); // 18 ✓
```

## 14.2 矩阵链乘法

```
问题描述：
给定 n 个矩阵 A₁, A₂, ..., Aₙ
矩阵 Aᵢ 的维度为 p[i-1] × p[i]
找到最优的乘法顺序，使得标量乘法次数最少
```

### 解法

```javascript
/**
 * 矩阵链乘法
 */
function matrixChainOrder(p) {
    const n = p.length - 1;
    
    // m[i][j] = 计算 Aᵢ...Aⱼ 的最小代价
    const m = Array.from({ length: n + 1 }, () => new Array(n + 1).fill(0));
    
    // s[i][j] = 最优断开位置
    const s = Array.from({ length: n + 1 }, () => new Array(n + 1).fill(0));
    
    // l 是链的长度
    for (let l = 2; l <= n; l++) {
        // i 是起始位置
        for (let i = 1; i <= n - l + 1; i++) {
            const j = i + l - 1;
            m[i][j] = Infinity;
            
            // k 是断开位置
            for (let k = i; k < j; k++) {
                const cost = m[i][k] + m[k + 1][j] + p[i - 1] * p[k] * p[j];
                
                if (cost < m[i][j]) {
                    m[i][j] = cost;
                    s[i][j] = k;
                }
            }
        }
    }
    
    return { m, s };
}

function printOptimalParens(s, i, j) {
    if (i === j) {
        return `A${i}`;
    } else {
        const k = s[i][j];
        return `(${printOptimalParens(s, i, k)}${printOptimalParens(s, k + 1, j)})`;
    }
}

// 测试
const dimensions = [30, 35, 15, 5, 10, 20, 25];
const result = matrixChainOrder(dimensions);
console.log('最小乘法次数:', result.m[1][6]); // 15125 ✓
console.log('最优加括号方式:', printOptimalParens(result.s, 1, 6)); 
// ((A₁(A₂A₃))((A₄A₅)A₆)) ✓
```

## 14.3 最长公共子序列

```
问题描述：
给定两个序列 X 和 Y
找到它们的最长公共子序列（LCS）
```

### 解法

```javascript
/**
 * 最长公共子序列
 */
function lcs(X, Y) {
    const m = X.length;
    const n = Y.length;
    
    // c[i][j] = X[0..i-1] 和 Y[0..j-1] 的 LCS 长度
    const c = Array.from({ length: m + 1 }, () => new Array(n + 1).fill(0));
    
    // b[i][j] = 记录方向
    const b = Array.from({ length: m + 1 }, () => new Array(n + 1).fill(0));
    
    for (let i = 1; i <= m; i++) {
        for (let j = 1; j <= n; j++) {
            if (X[i - 1] === Y[j - 1]) {
                c[i][j] = c[i - 1][j - 1] + 1;
                b[i][j] = '↖';
            } else if (c[i - 1][j] >= c[i][j - 1]) {
                c[i][j] = c[i - 1][j];
                b[i][j] = '↑';
            } else {
                c[i][j] = c[i][j - 1];
                b[i][j] = '←';
            }
        }
    }
    
    return { c, b };
}

function printLCS(b, X, i, j) {
    if (i === 0 || j === 0) return '';
    
    if (b[i][j] === '↖') {
        return printLCS(b, X, i - 1, j - 1) + X[i - 1];
    } else if (b[i][j] === '↑') {
        return printLCS(b, X, i - 1, j);
    } else {
        return printLCS(b, X, i, j - 1);
    }
}

// 测试
const X = 'ABCBDAB';
const Y = 'BDCABA';
const result = lcs(X, Y);
console.log('LCS 长度:', result.c[X.length][Y.length]); // 4 ✓
console.log('LCS:', printLCS(result.b, X, X.length, Y.length)); // BCBA ✓
```
