## 9.1 最小值和最大值

### 同时找最小值和最大值

```javascript
/**
 * 同时找出最小值和最大值
 * 比较次数：3n/2 - 2
 */
function findMinMax(arr) {
    if (arr.length === 0) return { min: null, max: null };
    
    let min, max;
    let i = 0;
    
    // 如果元素个数是奇数
    if (arr.length % 2 === 1) {
        min = max = arr[0];
        i = 1;
    } else {
        // 如果元素个数是偶数
        min = Math.min(arr[0], arr[1]);
        max = Math.max(arr[0], arr[1]);
        i = 2;
    }
    
    // 成对处理元素
    for (; i < arr.length; i += 2) {
        if (arr[i] < arr[i + 1]) {
            min = Math.min(min, arr[i]);
            max = Math.max(max, arr[i + 1]);
        } else {
            min = Math.min(min, arr[i + 1]);
            max = Math.max(max, arr[i]);
        }
    }
    
    return { min, max };
}

// 测试
const arr = [3, 5, 1, 9, 2, 8, 4, 7];
console.log(findMinMax(arr));
// 输出：{ min: 1, max: 9 } ✓
```

## 9.2 期望为线性的选择算法

### 随机化选择

```javascript
/**
 * 随机化选择算法 - 找第 i 小的元素
 * 期望时间复杂度：O(n)
 */
function randomizedSelect(arr, i) {
    return select(arr, 0, arr.length - 1, i);
}

function select(arr, left, right, i) {
    if (left === right) {
        return arr[left];
    }
    
    // 随机分区
    const pivotIndex = randomizedPartition(arr, left, right);
    const k = pivotIndex - left + 1;  // pivot 是第 k 小的元素
    
    if (i === k) {
        return arr[pivotIndex];
    } else if (i < k) {
        return select(arr, left, pivotIndex - 1, i);
    } else {
        return select(arr, pivotIndex + 1, right, i - k);
    }
}

function randomizedPartition(arr, left, right) {
    const randomIndex = Math.floor(Math.random() * (right - left + 1)) + left;
    [arr[randomIndex], arr[right]] = [arr[right], arr[randomIndex]];
    return partition(arr, left, right);
}

function partition(arr, left, right) {
    const pivot = arr[right];
    let i = left - 1;
    
    for (let j = left; j < right; j++) {
        if (arr[j] <= pivot) {
            i++;
            [arr[i], arr[j]] = [arr[j], arr[i]];
        }
    }
    
    [arr[i + 1], arr[right]] = [arr[right], arr[i + 1]];
    return i + 1;
}

// 测试
const arr = [7, 10, 4, 3, 20, 15];
console.log('第 3 小的元素:', randomizedSelect(arr, 3));
// 输出：7 ✓
```

## 9.3 最坏情况线性的选择算法

### BFPRT 算法（中位数的中位数）

```javascript
/**
 * BFPRT 算法 - 最坏情况 O(n)
 * 也叫"中位数的中位数"算法
 */
function medianOfMediansSelect(arr, i) {
    return bfprt(arr, 0, arr.length - 1, i);
}

function bfprt(arr, left, right, i) {
    if (left === right) {
        return arr[left];
    }
    
    // 找到一个好的 pivot
    const pivotIndex = medianOfMedians(arr, left, right);
    
    // 分区
    const k = partitionAroundPivot(arr, left, right, pivotIndex) - left + 1;
    
    if (i === k) {
        return arr[pivotIndex];
    } else if (i < k) {
        return bfprt(arr, left, pivotIndex - 1, i);
    } else {
        return bfprt(arr, pivotIndex + 1, right, i - k);
    }
}

function medianOfMedians(arr, left, right) {
    const n = right - left + 1;
    const numGroups = Math.ceil(n / 5);
    const medians = [];
    
    // 将数组分成 5 个一组
    for (let g = 0; g < numGroups; g++) {
        const groupLeft = left + g * 5;
        const groupRight = Math.min(groupLeft + 4, right);
        
        // 对每组排序，找中位数
        const group = arr.slice(groupLeft, groupRight + 1);
        group.sort((a, b) => a - b);
        
        const medianIndex = Math.floor(group.length / 2);
        medians.push(group[medianIndex]);
    }
    
    // 递归找中位数的中位数
    if (medians.length <= 5) {
        medians.sort((a, b) => a - b);
        return medians[Math.floor(medians.length / 2)];
    } else {
        return medianOfMedians(medians, 0, medians.length - 1, 
                              Math.floor(medians.length / 2) + 1);
    }
}

function partitionAroundPivot(arr, left, right, pivotIndex) {
    const pivotValue = arr[pivotIndex];
    [arr[pivotIndex], arr[right]] = [arr[right], arr[pivotIndex]];
    
    let storeIndex = left;
    for (let i = left; i < right; i++) {
        if (arr[i] < pivotValue) {
            [arr[i], arr[storeIndex]] = [arr[storeIndex], arr[i]];
            storeIndex++;
        }
    }
    
    [arr[storeIndex], arr[right]] = [arr[right], arr[storeIndex]];
    return storeIndex;
}

// 最坏情况时间复杂度：O(n)
```
