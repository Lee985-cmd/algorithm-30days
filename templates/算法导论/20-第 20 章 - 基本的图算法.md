## 20.1 图的表示

```javascript
/**
 * 图的邻接表表示
 */
class Graph {
    constructor(vertices, directed = false) {
        this.vertices = vertices;
        this.directed = directed;
        this.adj = new Array(vertices).fill(null).map(() => []);
    }
    
    addEdge(u, v) {
        this.adj[u].push(v);
        if (!this.directed) {
            this.adj[v].push(u);
        }
    }
}
```

## 20.2 广度优先搜索（BFS）

```javascript
/**
 * BFS - 最短路径（边数最少）
 */
function bfs(graph, s) {
    const visited = new Array(graph.vertices).fill(false);
    const dist = new Array(graph.vertices).fill(Infinity);
    const parent = new Array(graph.vertices).fill(null);
    
    visited[s] = true;
    dist[s] = 0;
    
    const queue = [s];
    
    while (queue.length > 0) {
        const u = queue.shift();
        
        for (const v of graph.adj[u]) {
            if (!visited[v]) {
                visited[v] = true;
                dist[v] = dist[u] + 1;
                parent[v] = u;
                queue.push(v);
            }
        }
    }
    
    return { dist, parent };
}

// 测试
const g = new Graph(6);
g.addEdge(0, 1);
g.addEdge(0, 2);
g.addEdge(1, 3);
g.addEdge(2, 3);
g.addEdge(3, 4);
g.addEdge(4, 5);

const result = bfs(g, 0);
console.log('从 0 到各点的最短距离:', result.dist);
```

## 20.3 深度优先搜索（DFS）

```javascript
/**
 * DFS
 */
function dfs(graph) {
    const visited = new Array(graph.vertices).fill(false);
    const discovery = new Array(graph.vertices).fill(0);
    const finish = new Array(graph.vertices).fill(0);
    const parent = new Array(graph.vertices).fill(null);
    let time = 0;
    
    function dfsVisit(u) {
        time++;
        discovery[u] = time;
        visited[u] = true;
        
        for (const v of graph.adj[u]) {
            if (!visited[v]) {
                parent[v] = u;
                dfsVisit(v);
            }
        }
        
        time++;
        finish[u] = time;
    }
    
    for (let i = 0; i < graph.vertices; i++) {
        if (!visited[i]) {
            dfsVisit(i);
        }
    }
    
    return { discovery, finish, parent };
}

// 测试
const result = dfs(g);
console.log('发现时间:', result.discovery);
console.log('完成时间:', result.finish);
```
