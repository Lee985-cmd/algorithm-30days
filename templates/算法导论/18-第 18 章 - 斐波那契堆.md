# 📚 算法导论第 18 章 - 斐波那契堆

## 18.1 斐波那契堆结构

```
斐波那契堆是一种可并堆，支持以下操作：
- INSERT: O(1) 摊还时间
- MINIMUM: O(1) 摊还时间
- UNION: O(1) 摊还时间
- EXTRACT-MIN: O(log n) 摊还时间
- DECREASE-KEY: O(1) 摊还时间
- DELETE: O(log n) 摊还时间

应用：Dijkstra 算法、Prim 算法的优化
```

## 18.2 核心操作

```javascript
/**
 * 斐波那契堆节点
 */
class FibNode {
    constructor(key) {
        this.key = key;
        this.degree = 0;
        this.mark = false;
        this.parent = null;
        this.child = null;
        this.left = this;
        this.right = this;
    }
}

/**
 * 斐波那契堆
 */
class FibonacciHeap {
    constructor() {
        this.min = null;
        this.n = 0;
    }
    
    insert(key) {
        const node = new FibNode(key);
        
        if (this.min === null) {
            this.min = node;
        } else {
            this._linkRoot(node);
            
            if (node.key < this.min.key) {
                this.min = node;
            }
        }
        
        this.n++;
        return node;
    }
    
    minimum() {
        return this.min;
    }
    
    union(other) {
        if (other.min === null) return this;
        if (this.min === null) {
            this.min = other.min;
        } else {
            // 连接两个根表
            const temp = this.min.right;
            this.min.right = other.min;
            other.min.left.right = temp;
            temp.left = other.min.left;
            other.min.left = this.min;
            
            if (other.min.key < this.min.key) {
                this.min = other.min;
            }
        }
        
        this.n += other.n;
        return this;
    }
    
    extractMin() {
        const z = this.min;
        
        if (z !== null) {
            // 将所有孩子加入根表
            if (z.child !== null) {
                let child = z.child;
                do {
                    const next = child.right;
                    this._linkRoot(child);
                    child.parent = null;
                    child = next;
                } while (child !== z.child);
            }
            
            // 移除 z
            z.left.right = z.right;
            z.right.left = z.left;
            
            if (z === z.right) {
                this.min = null;
            } else {
                this.min = z.right;
                this._consolidate();
            }
            
            this.n--;
        }
        
        return z;
    }
    
    decreaseKey(node, newKey) {
        if (newKey > node.key) {
            throw new Error('新键值必须更小');
        }
        
        node.key = newKey;
        const parent = node.parent;
        
        if (parent !== null && node.key < parent.key) {
            this._cut(node, parent);
            this._cascadingCut(parent);
        }
        
        if (node.key < this.min.key) {
            this.min = node;
        }
    }
    
    delete(node) {
        this.decreaseKey(node, -Infinity);
        this.extractMin();
    }
    
    _linkRoot(node) {
        if (this.min === null) {
            node.left = node;
            node.right = node;
        } else {
            node.right = this.min.right;
            node.left = this.min;
            this.min.right.left = node;
            this.min.right = node;
        }
    }
    
    _consolidate() {
        const A = [];
        const nodes = [];
        
        // 收集所有根节点
        let current = this.min;
        do {
            nodes.push(current);
            current = current.right;
        } while (current !== this.min);
        
        // 合并相同度数的树
        for (const node of nodes) {
            let x = node;
            let d = x.degree;
            
            while (A[d] !== undefined) {
                const y = A[d];
                
                if (x.key > y.key) {
                    [x, y] = [y, x];
                }
                
                this._link(y, x);
                A[d] = undefined;
                d++;
            }
            
            A[d] = x;
        }
        
        // 重建根表
        this.min = null;
        for (const node of A) {
            if (node !== undefined) {
                if (this.min === null) {
                    this.min = node;
                } else if (node.key < this.min.key) {
                    this.min = node;
                }
            }
        }
    }
    
    _link(y, x) {
        // 将 y 链接为 x 的孩子
        y.left.right = y.right;
        y.right.left = y.left;
        
        if (x.child === null) {
            x.child = y;
            y.parent = x;
            y.left = y;
            y.right = y;
        } else {
            const child = x.child;
            y.right = child.right;
            y.left = child;
            child.right.left = y;
            child.right = y;
            y.parent = x;
        }
        
        x.degree++;
        y.mark = false;
    }
    
    _cut(x, y) {
        // 从 y 的孩子中移除 x
        if (y.degree === 1) {
            y.child = null;
        } else {
            y.child = x.right;
        }
        
        x.left.right = x.right;
        x.right.left = x.left;
        
        this._linkRoot(x);
        x.parent = null;
        y.degree--;
    }
    
    _cascadingCut(y) {
        const parent = y.parent;
        
        if (parent !== null) {
            if (!y.mark) {
                y.mark = true;
            } else {
                this._cut(y, parent);
                this._cascadingCut(parent);
            }
        }
    }
}

// 测试
const heap = new FibonacciHeap();
heap.insert(5);
heap.insert(3);
heap.insert(7);
console.log('最小值:', heap.minimum().key); // 3 ✓
```
