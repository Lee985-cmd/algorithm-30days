# 📚 算法导论第 17 章 - B 树

## 17.1 B 树的定义

```
B 树是一种自平衡的多路搜索树，适合磁盘存储

性质：
1. 每个节点最多有 m 个孩子
2. 根节点至少有 2 个孩子（除非是叶子）
3. 非根内部节点至少有 ⌈m/2⌉ 个孩子
4. 所有叶子都在同一层
5. 关键字在节点内有序

应用：数据库索引、文件系统
```

## 17.2 B 树操作

```javascript
/**
 * B 树节点
 */
class BTreeNode {
    constructor(t, leaf) {
        this.t = t;  // 最小度数
        this.leaf = leaf;
        this.keys = new Array(2 * t - 1).fill(null);
        this.children = new Array(2 * t).fill(null);
        this.numKeys = 0;
    }
}

/**
 * B 树
 */
class BTree {
    constructor(t) {
        this.t = t;  // 最小度数
        this.root = null;
    }
    
    search(k, node = this.root) {
        if (node === null) return null;
        
        let i = 0;
        while (i < node.numKeys && k > node.keys[i]) {
            i++;
        }
        
        if (i < node.numKeys && k === node.keys[i]) {
            return node;
        }
        
        if (node.leaf) {
            return null;
        } else {
            return this.search(k, node.children[i]);
        }
    }
    
    insert(k) {
        if (this.root === null) {
            this.root = new BTreeNode(this.t, true);
            this.root.keys[0] = k;
            this.root.numKeys = 1;
        } else {
            if (this.root.numKeys === 2 * this.t - 1) {
                const newRoot = new BTreeNode(this.t, false);
                newRoot.children[0] = this.root;
                this.splitChild(newRoot, 0);
                this.insertNonFull(newRoot, k);
                this.root = newRoot;
            } else {
                this.insertNonFull(this.root, k);
            }
        }
    }
    
    splitChild(parent, i) {
        const t = this.t;
        const child = parent.children[i];
        const newNode = new BTreeNode(t, child.leaf);
        
        newNode.numKeys = t - 1;
        
        for (let j = 0; j < t - 1; j++) {
            newNode.keys[j] = child.keys[j + t];
        }
        
        if (!child.leaf) {
            for (let j = 0; j < t; j++) {
                newNode.children[j] = child.children[j + t];
            }
        }
        
        child.numKeys = t - 1;
        
        for (let j = parent.numKeys; j >= i + 1; j--) {
            parent.children[j + 1] = parent.children[j];
        }
        
        parent.children[i + 1] = newNode;
        
        for (let j = parent.numKeys - 1; j >= i; j--) {
            parent.keys[j + 1] = parent.keys[j];
        }
        
        parent.keys[i] = child.keys[t - 1];
        parent.numKeys++;
    }
    
    insertNonFull(node, k) {
        let i = node.numKeys - 1;
        
        if (node.leaf) {
            while (i >= 0 && k < node.keys[i]) {
                node.keys[i + 1] = node.keys[i];
                i--;
            }
            node.keys[i + 1] = k;
            node.numKeys++;
        } else {
            while (i >= 0 && k < node.keys[i]) {
                i--;
            }
            i++;
            
            if (node.children[i].numKeys === 2 * this.t - 1) {
                this.splitChild(node, i);
                if (k > node.keys[i]) {
                    i++;
                }
            }
            
            this.insertNonFull(node.children[i], k);
        }
    }
}

// 测试
const tree = new BTree(3);
tree.insert(10);
tree.insert(20);
tree.insert(5);
tree.insert(6);
console.log('B 树创建成功！');
```
