## 23.1 Floyd-Warshall 算法

```javascript
/**
 * Floyd-Warshall 算法 - 所有点对最短路径
 */
function floydWarshall(graph) {
    const n = graph.vertices;
    
    // 初始化距离矩阵
    const dist = Array.from({ length: n }, (_, i) => 
        Array.from({ length: n }, (_, j) => {
            if (i === j) return 0;
            if (graph.adj[i].includes(j)) return 1;  // 假设有边权重为 1
            return Infinity;
        })
    );
    
    // 动态规划
    for (let k = 0; k < n; k++) {
        for (let i = 0; i < n; i++) {
            for (let j = 0; j < n; j++) {
                if (dist[i][k] !== Infinity && dist[k][j] !== Infinity) {
                    dist[i][j] = Math.min(dist[i][j], dist[i][k] + dist[k][j]);
                }
            }
        }
    }
    
    return dist;
}

// 测试
const g = new Graph(4, true);  // 有向图
g.addEdge(0, 1);
g.addEdge(0, 3);
g.addEdge(1, 0);
g.addEdge(1, 2);
g.addEdge(2, 0);
g.addEdge(2, 3);
g.addEdge(3, 1);

const result = floydWarshall(g);
console.log('Floyd-Warshall 结果:');
result.forEach(row => console.log(row));
```
