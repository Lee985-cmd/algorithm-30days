## 28.1 多项式表示

```
多项式的两种表示方法：

1. 系数表示：
   A(x) = a₀ + a₁x + a₂x² + ... + aₙ₋₁xⁿ⁻¹
   表示为向量：(a₀, a₁, ..., aₙ₋₁)

2. 点值表示：
   n-1 次多项式由 n 个点值对唯一确定
   {(x₀, y₀), (x₁, y₁), ..., (xₙ₋₁, yₙ₋₁)}
   其中 yᵢ = A(xᵢ)
```

## 28.2 快速傅里叶变换（FFT）

```javascript
/**
 * FFT - 快速傅里叶变换
 * 时间复杂度：O(n log n)
 */
function fft(a, invert) {
    const n = a.length;
    
    if (n === 1) return;
    
    // 分治
    const a0 = new Array(n / 2);
    const a1 = new Array(n / 2);
    
    for (let i = 0; i < n / 2; i++) {
        a0[i] = a[2 * i];
        a1[i] = a[2 * i + 1];
    }
    
    fft(a0, invert);
    fft(a1, invert);
    
    // 合并
    const angle = (2 * Math.PI / n) * (invert ? -1 : 1);
    const wn = { 
        real: Math.cos(angle), 
        imag: Math.sin(angle) 
    };
    
    let w = { real: 1, imag: 0 };
    
    for (let i = 0; i < n / 2; i++) {
        // 复数乘法
        const t = {
            real: w.real * a1[i].real - w.imag * a1[i].imag,
            imag: w.real * a1[i].imag + w.imag * a1[i].real
        };
        
        a[i] = {
            real: a0[i].real + t.real,
            imag: a0[i].imag + t.imag
        };
        
        a[i + n / 2] = {
            real: a0[i].real - t.real,
            imag: a0[i].imag - t.imag
        };
        
        // w *= wn
        const newW = {
            real: w.real * wn.real - w.imag * wn.imag,
            imag: w.real * wn.imag + w.imag * wn.real
        };
        w = newW;
    }
}

/**
 * 多项式乘法
 */
function multiplyPolynomials(A, B) {
    const n = 1;
    while (n < A.length + B.length) n *= 2;
    
    // 补齐到 2 的幂
    const fa = A.map(x => ({ real: x, imag: 0 }));
    const fb = B.map(x => ({ real: x, imag: 0 }));
    
    while (fa.length < n) fa.push({ real: 0, imag: 0 });
    while (fb.length < n) fb.push({ real: 0, imag: 0 });
    
    // FFT
    fft(fa, false);
    fft(fb, false);
    
    // 点乘
    const fc = fa.map((val, i) => ({
        real: val.real * fb[i].real - val.imag * fb[i].imag,
        imag: val.real * fb[i].imag + val.imag * fb[i].real
    }));
    
    // IFFT
    fft(fc, true);
    
    // 取实部并归一化
    const result = fc.map(val => Math.round(val.real / n));
    
    return result;
}

// 测试
const A = [1, 2, 3];  // 1 + 2x + 3x²
const B = [4, 5];     // 4 + 5x

const C = multiplyPolynomials(A, B);
console.log('多项式乘法结果:', C);
// [4, 13, 22, 15] ✓
// 即：4 + 13x + 22x² + 15x³
```
