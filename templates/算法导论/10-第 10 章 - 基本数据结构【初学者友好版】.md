## 🗄️ 栈（Stack）

### 💡 把栈想象成...

```
栈就像一摞盘子：
- 只能从最上面放（push）
- 只能从最上面拿（pop）
- 后放的先拿（LIFO: Last In First Out）

生活例子：
- 浏览器的后退按钮
- Word 的撤销操作
- 弹夹装子弹
```

### 代码实现

```javascript
/**
 * 栈的实现
 */
class Stack {
    constructor() {
        this.items = [];
    }
    
    /**
     * 入栈（放盘子上面）
     */
    push(element) {
        this.items.push(element);
    }
    
    /**
     * 出栈（从盘子顶拿）
     */
    pop() {
        if (this.isEmpty()) return null;
        return this.items.pop();
    }
    
    /**
     * 查看栈顶元素
     */
    peek() {
        if (this.isEmpty()) return null;
        return this.items[this.items.length - 1];
    }
    
    /**
     * 判断是否为空
     */
    isEmpty() {
        return this.items.length === 0;
    }
    
    /**
     * 返回栈的大小
     */
    size() {
        return this.items.length;
    }
}

// 🧪 测试
const stack = new Stack();
stack.push(1);
stack.push(2);
stack.push(3);
console.log(stack.peek());   // 3
console.log(stack.pop());    // 3
console.log(stack.pop());    // 2
console.log(stack.size());   // 1

// ⏱️ 时间复杂度：都是 O(1)
```

### 应用场景

```javascript
// 应用 1: 括号匹配
function isValidParentheses(s) {
    const stack = new Stack();
    const map = {
        ')': '(',
        ']': '[',
        '}': '{'
    };
    
    for (let char of s) {
        if (char in map) {
            const top = stack.pop();
            if (map[char] !== top) return false;
        } else {
            stack.push(char);
        }
    }
    
    return stack.isEmpty();
}

// 应用 2: 浏览器后退
function browserHistory() {
    const backStack = new Stack();
    const forwardStack = new Stack();
    let currentPage = 'home';
    
    function visit(page) {
        backStack.push(currentPage);
        currentPage = page;
        forwardStack.clear();  // 清空前进历史
    }
    
    function goBack() {
        if (!backStack.isEmpty()) {
            forwardStack.push(currentPage);
            currentPage = backStack.pop();
        }
    }
    
    function goForward() {
        if (!forwardStack.isEmpty()) {
            backStack.push(currentPage);
            currentPage = forwardStack.pop();
        }
    }
}
```

## 🔄 队列（Queue）

### 💡 把队列想象成...

```
队列就像排队买票：
- 新来的排后面（enqueue）
- 前面的先买（dequeue）
- 先来先服务（FIFO: First In First Out）

生活例子：
- 打印任务队列
- 餐厅等位
- 消息队列
```

### 代码实现

```javascript
/**
 * 队列的实现
 */
class Queue {
    constructor() {
        this.items = [];
        this.frontIndex = 0;
    }
    
    /**
     * 入队（排到队尾）
     */
    enqueue(element) {
        this.items.push(element);
    }
    
    /**
     * 出队（队头离开）
     */
    dequeue() {
        if (this.isEmpty()) return null;
        const item = this.items[this.frontIndex];
        this.frontIndex++;
        return item;
    }
    
    /**
     * 查看队头元素
     */
    front() {
        if (this.isEmpty()) return null;
        return this.items[this.frontIndex];
    }
    
    /**
     * 判断是否为空
     */
    isEmpty() {
        return this.frontIndex >= this.items.length;
    }
    
    /**
     * 返回队列大小
     */
    size() {
        return this.items.length - this.frontIndex;
    }
}

// 🧪 测试
const queue = new Queue();
queue.enqueue(1);
queue.enqueue(2);
queue.enqueue(3);
console.log(queue.front());   // 1
console.log(queue.dequeue()); // 1
console.log(queue.dequeue()); // 2
console.log(queue.size());    // 1

// ⏱️ 时间复杂度：都是 O(1)
```

## 🔗 链表（Linked List）

### 💡 把链表想象成...

```
链表就像寻宝游戏：
- 每个宝藏有一张纸条，写着下一个宝藏的位置
- 要找第 n 个宝藏，必须从第 1 个开始找
- 不能直接跳到第 n 个（不像数组可以随机访问）

vs 数组：
数组：电影院座位，可以直接走到第几排
链表：寻宝游戏，必须按线索一个一个找
```

### 代码实现

```javascript
/**
 * 链表节点
 */
class ListNode {
    constructor(val) {
        this.val = val;
        this.next = null;
    }
}

/**
 * 单链表
 */
class LinkedList {
    constructor() {
        this.head = null;
        this.size = 0;
    }
    
    /**
     * 在头部插入
     */
    insertAtHead(val) {
        const node = new ListNode(val);
        node.next = this.head;
        this.head = node;
        this.size++;
    }
    
    /**
     * 删除指定值的节点
     */
    delete(val) {
        if (!this.head) return false;
        
        if (this.head.val === val) {
            this.head = this.head.next;
            this.size--;
            return true;
        }
        
        let current = this.head;
        while (current.next) {
            if (current.next.val === val) {
                current.next = current.next.next;
                this.size--;
                return true;
            }
            current = current.next;
        }
        
        return false;
    }
    
    /**
     * 查找指定值
     */
    search(val) {
        let current = this.head;
        let index = 0;
        
        while (current) {
            if (current.val === val) {
                return index;
            }
            current = current.next;
            index++;
        }
        
        return -1;
    }
    
    /**
     * 反转链表
     */
    reverse() {
        let prev = null;
        let current = this.head;
        
        while (current) {
            const next = current.next;  // 保存下一个
            current.next = prev;         // 反转指针
            prev = current;              // 移动 prev
            current = next;              // 移动 current
        }
        
        this.head = prev;
    }
}

// 🧪 测试
const list = new LinkedList();
list.insertAtHead(3);
list.insertAtHead(2);
list.insertAtHead(1);
console.log(list.search(2));  // 1
list.delete(2);
console.log(list.search(2));  // -1
list.reverse();

// ⏱️ 时间复杂度：
// 头部插入/删除：O(1)
// 查找/删除中间：O(n)
```

### ❗ 常见错误

```javascript
// ❌ 错误 1: 忘记更新 head
function deleteHead(list) {
    list.head = list.head.next;
    // 忘了 list.size--
}

// ❌ 错误 2: 断链
function reverseWrong(head) {
    let prev = null;
    let current = head;
    
    while (current) {
        // ❌ 没有保存 next，会丢失后面的节点！
        current.next = prev;
        prev = current;
        current = current.next;  // current 已经是 prev 了！
    }
}
```

## 📊 对比总结

| 数据结构 | 特点 | 访问 | 插入/删除 | 应用 |
|---------|------|------|----------|------|
| 栈 | LIFO | O(1) | O(1) | 撤销、括号匹配 |
| 队列 | FIFO | O(1) | O(1) | BFS、任务调度 |
| 链表 | 顺序访问 | O(n) | O(1) 头部 | 动态集合 |
| 数组 | 随机访问 | O(1) | O(n) | 一般情况 |
