## 29.2 欧几里得算法

```javascript
/**
 * 欧几里得算法 - 求最大公约数
 */
function gcd(a, b) {
    if (b === 0) return a;
    return gcd(b, a % b);
}

// 迭代版本
function gcdIterative(a, b) {
    while (b !== 0) {
        const temp = b;
        b = a % b;
        a = temp;
    }
    return a;
}

// 测试
console.log('gcd(48, 18):', gcd(48, 18)); // 6 ✓
console.log('gcd(100, 25):', gcd(100, 25)); // 25 ✓
```

## 29.3 扩展欧几里得算法

```javascript
/**
 * 扩展欧几里得算法
 * 返回 {g, x, y} 使得 ax + by = g = gcd(a, b)
 */
function extendedGCD(a, b) {
    if (b === 0) {
        return { g: a, x: 1, y: 0 };
    }
    
    const { g, x, y } = extendedGCD(b, a % b);
    
    return {
        g,
        x: y,
        y: x - Math.floor(a / b) * y
    };
}

// 测试
const result = extendedGCD(35, 15);
console.log('extendedGCD(35, 15):', result);
// 35*(-1) + 15*3 = 10 ✓
```

## 29.4 模线性方程求解

```javascript
/**
 * 求解 ax ≡ b (mod n)
 */
function modLinearEquation(a, b, n) {
    const { g, x, y } = extendedGCD(a, n);
    
    if (b % g !== 0) {
        return [];  // 无解
    }
    
    const x0 = (x * (b / g)) % n;
    const solutions = [];
    
    for (let i = 0; i < g; i++) {
        solutions.push((x0 + i * (n / g)) % n);
    }
    
    return solutions;
}

// 测试
console.log('求解 3x ≡ 4 (mod 7):', modLinearEquation(3, 4, 7));
// [6] ✓ 因为 3*6 = 18 ≡ 4 (mod 7)
```

## 29.5 中国剩余定理

```javascript
/**
 * 中国剩余定理
 * 求解同余方程组：
 * x ≡ a₁ (mod m₁)
 * x ≡ a₂ (mod m₂)
 * ...
 * x ≡ aₖ (mod mₖ)
 */
function chineseRemainderTheorem(remainders, moduli) {
    const k = remainders.length;
    
    // 计算所有模数的乘积
    let M = 1;
    for (const m of moduli) {
        M *= m;
    }
    
    // 计算每个 Mi = M / mi
    const Mi = moduli.map(m => M / m);
    
    // 计算每个 Mi 的模逆
    const yi = Mi.map((val, i) => {
        const { x } = extendedGCD(val, moduli[i]);
        return x;
    });
    
    // 计算最终结果
    let x = 0;
    for (let i = 0; i < k; i++) {
        x += remainders[i] * Mi[i] * yi[i];
    }
    
    return ((x % M) + M) % M;
}

// 测试
// x ≡ 2 (mod 3)
// x ≡ 3 (mod 4)
// x ≡ 1 (mod 5)
const result = chineseRemainderTheorem([2, 3, 1], [3, 4, 5]);
console.log('中国剩余定理解:', result); // 11 ✓
// 验证：11 % 3 = 2, 11 % 4 = 3, 11 % 5 = 1 ✓
```

## 29.6 素性测试

```javascript
/**
 * Miller-Rabin 素性测试
 */
function millerRabin(n, k = 5) {
    if (n <= 1) return false;
    if (n <= 3) return true;
    if (n % 2 === 0) return false;
    
    // 将 n-1 写成 2^r * d
    let r = 0;
    let d = n - 1;
    while (d % 2 === 0) {
        r++;
        d /= 2;
    }
    
    // 进行 k 次测试
    for (let i = 0; i < k; i++) {
        if (!millerRabinTest(n, d, r)) {
            return false;
        }
    }
    
    return true;
}

function millerRabinTest(n, d, r) {
    const a = 2 + Math.floor(Math.random() * (n - 3));
    let x = modularExponentiation(a, d, n);
    
    if (x === 1 || x === n - 1) return true;
    
    for (let j = 0; j < r - 1; j++) {
        x = modularExponentiation(x, 2, n);
        if (x === n - 1) return true;
    }
    
    return false;
}

function modularExponentiation(base, exp, mod) {
    let result = 1;
    base = base % mod;
    
    while (exp > 0) {
        if (exp % 2 === 1) {
            result = (result * base) % mod;
        }
        exp = exp >> 1;
        base = (base * base) % mod;
    }
    
    return result;
}

// 测试
console.log('17 是素数吗？', millerRabin(17)); // true ✓
console.log('20 是素数吗？', millerRabin(20)); // false ✓
```

## 29.7 RSA 公钥密码系统

```javascript
/**
 * RSA 加密解密
 */
class RSA {
    constructor(p, q) {
        this.p = p;
        this.q = q;
        this.n = p * q;
        this.phi = (p - 1) * (q - 1);
        this.e = this.findE();
        this.d = this.findD();
    }
    
    findE() {
        for (let e = 2; e < this.phi; e++) {
            if (gcd(e, this.phi) === 1) {
                return e;
            }
        }
    }
    
    findD() {
        const { x } = extendedGCD(this.e, this.phi);
        return ((x % this.phi) + this.phi) % this.phi;
    }
    
    encrypt(message) {
        return modularExponentiation(message, this.e, this.n);
    }
    
    decrypt(cipher) {
        return modularExponentiation(cipher, this.d, this.n);
    }
}

// 测试
const rsa = new RSA(61, 53);
console.log('公钥 (e, n):', `(${rsa.e}, ${rsa.n})`);
console.log('私钥 (d, n):', `(${rsa.d}, ${rsa.n})`);

const message = 65;
const encrypted = rsa.encrypt(message);
const decrypted = rsa.decrypt(encrypted);

console.log('原文:', message);
console.log('密文:', encrypted);
console.log('解密:', decrypted); // 65 ✓
```
