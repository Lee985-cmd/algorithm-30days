## 27.1 旅行商问题的近似算法

```javascript
/**
 * TSP 的 2-近似算法（基于 MST）
 */
function tspApprox(vertices, edges) {
    // 1. 构建最小生成树
    const mst = kruskalMST(vertices, edges);
    
    // 2. 前序遍历 MST
    const adj = buildAdjList(vertices, mst.mst);
    const visited = new Set();
    const tour = [];
    
    function preorder(u) {
        if (visited.has(u)) return;
        visited.add(u);
        tour.push(u);
        
        for (const v of adj[u]) {
            preorder(v);
        }
    }
    
    preorder(0);
    tour.push(0);  // 回到起点
    
    return tour;
}

function buildAdjList(n, edges) {
    const adj = Array.from({ length: n }, () => []);
    for (const edge of edges) {
        adj[edge.u].push(edge.v);
        adj[edge.v].push(edge.u);
    }
    return adj;
}

// 这是一个 2-近似算法
// 代价不超过最优解的 2 倍
```

## 27.2 集合覆盖问题

```javascript
/**
 * 集合覆盖的贪心近似算法
 * 近似比：H(n) ≈ ln n
 */
function greedySetCover(universe, subsets) {
    const U = new Set(universe);
    const covered = new Set();
    const result = [];
    
    while (covered.size < U.size) {
        // 选择能覆盖最多未覆盖元素的集合
        let bestSet = null;
        let bestCount = 0;
        
        for (let i = 0; i < subsets.length; i++) {
            const set = subsets[i];
            let count = 0;
            
            for (const elem of set) {
                if (!covered.has(elem) && U.has(elem)) {
                    count++;
                }
            }
            
            if (count > bestCount) {
                bestCount = count;
                bestSet = i;
            }
        }
        
        if (bestSet === null) break;
        
        result.push(bestSet);
        for (const elem of subsets[bestSet]) {
            covered.add(elem);
        }
    }
    
    return result;
}

// 测试
const universe = [1, 2, 3, 4, 5];
const subsets = [
    [1, 2, 3],
    [2, 4],
    [3, 4],
    [4, 5]
];

const result = greedySetCover(universe, subsets);
console.log('选择的集合:', result); // [0, 3] ✓
```
