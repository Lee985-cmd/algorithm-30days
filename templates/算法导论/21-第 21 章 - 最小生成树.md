## 21.1 Prim 算法

```javascript
/**
 * Prim 算法 - 最小生成树
 */
function primMST(graph, weights, start = 0) {
    const n = graph.vertices;
    const key = new Array(n).fill(Infinity);
    const parent = new Array(n).fill(null);
    const inMST = new Array(n).fill(false);
    
    key[start] = 0;
    
    for (let count = 0; count < n; count++) {
        // 找最小的 key
        let u = -1;
        let minKey = Infinity;
        
        for (let v = 0; v < n; v++) {
            if (!inMST[v] && key[v] < minKey) {
                minKey = key[v];
                u = v;
            }
        }
        
        if (u === -1) break;
        
        inMST[u] = true;
        
        // 更新相邻顶点
        for (const v of graph.adj[u]) {
            if (!inMST[v] && weights[`${u}-${v}`] < key[v]) {
                key[v] = weights[`${u}-${v}`];
                parent[v] = u;
            }
        }
    }
    
    return { parent, key };
}

// 测试
const g = new Graph(5);
g.addEdge(0, 1);
g.addEdge(0, 3);
g.addEdge(1, 2);
g.addEdge(1, 3);
g.addEdge(1, 4);
g.addEdge(2, 4);
g.addEdge(3, 4);

const weights = {
    '0-1': 2, '0-3': 6,
    '1-2': 3, '1-3': 8, '1-4': 5,
    '2-4': 7,
    '3-4': 9
};

const result = primMST(g, weights);
console.log('Prim 算法的 MST:', result.parent);
```

## 21.2 Kruskal 算法

```javascript
/**
 * Kruskal 算法 - 最小生成树
 */
function kruskalMST(vertices, edges) {
    // 按权重排序
    edges.sort((a, b) => a.weight - b.weight);
    
    const uf = new UnionFind(vertices);
    const mst = [];
    let totalWeight = 0;
    
    for (const edge of edges) {
        if (uf.union(edge.u, edge.v)) {
            mst.push(edge);
            totalWeight += edge.weight;
            
            if (mst.length === vertices - 1) break;
        }
    }
    
    return { mst, totalWeight };
}

class UnionFind {
    constructor(n) {
        this.parent = new Array(n);
        for (let i = 0; i < n; i++) this.parent[i] = i;
    }
    
    find(x) {
        if (this.parent[x] !== x) {
            this.parent[x] = this.find(this.parent[x]);
        }
        return this.parent[x];
    }
    
    union(x, y) {
        const rootX = this.find(x);
        const rootY = this.find(y);
        if (rootX !== rootY) {
            this.parent[rootX] = rootY;
            return true;
        }
        return false;
    }
}

// 测试
const edges = [
    { u: 0, v: 1, weight: 2 },
    { u: 0, v: 3, weight: 6 },
    { u: 1, v: 2, weight: 3 },
    { u: 1, v: 3, weight: 8 },
    { u: 1, v: 4, weight: 5 },
    { u: 2, v: 4, weight: 7 },
    { u: 3, v: 4, weight: 9 }
];

const result = kruskalMST(5, edges);
console.log('Kruskal 算法的 MST:', result.mst);
console.log('总权重:', result.totalWeight); // 16 ✓
```
